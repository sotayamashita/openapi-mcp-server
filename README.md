{
  "mcpServer": {

      "command": "bun",
      "args": [
        "/aboopenapi-mcp-server/src/index.ts",
        // OpenAPI/Swagger spec URL or absolute file path
        "/aboopenapi-mcp-server/demo/openapi.yml"
      ]
    }
  }
}
